# Prime numbers in an interval

lower = int(input("Enter lower range: "))
upper = int(input("Enter upper range: "))

for num in range(lower, upper + 1):
    if num > 1:
        for i in range(2, num):
            if (num % i) == 0:
                break
        else:
            print(num)

# input

Enter lower range: 2
Enter upper range: 7

# output

2
3
5
7
