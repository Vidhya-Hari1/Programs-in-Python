# Multiplication Table

n = int(input("Enter any number"))
for i in range(1,11):
    print(str(n) + ' x ' + str(i) + ' = ' + str(n*i) )


# input

Enter any number: 4

# output

4 x 1 = 4
4 x 2 = 8
4 x 3 = 12
4 x 4 = 16
4 x 5 = 20
